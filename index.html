jQuery Noisy Plugin Demo

Fork on github and like on Forrst.

$("#default").noisy();
$("#opacity").noisy({ opacity : .5 });
$("#color").noisy({ color : 'rgb(250, 20, 20)' });
$("#monochromatic").noisy({ monochromatic : true });
$("#brightness").noisy({ brightness : 120 });
$("#multiple").noisy({ opacity : .3, color: 'rgb(40, 200, 40)', monochromatic: false, brightness: 100 });
